SELECT `imie`, `nazwisko` FROM `matematycy` WHERE `rok_urodzenia` > 1700;

SELECT `id`, `nazwisko` FROM `matematycy` WHERE `liczba_publikacji` > 10 AND liczba_publikacji < 30;

SELECT SUM(liczba_publikacji) AS liczba_publikacji  FROM `matematycy` WHERE rok_urodzenia > 1750;

CREATE TABLE `arytmetyka`.`publikacje` ( `id` INT NOT NULL AUTO_INCREMENT , `tytul` TEXT NOT NULL , `rok` MEDIUMINT NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;
